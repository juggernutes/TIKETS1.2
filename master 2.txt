/* ============================================================
   SCHEMA: core
   ORDEN: rol → permiso → area → puesto → sucursal → empleado
          → usuario → rol_permiso (FK diferida) → login
          → password_resets → folio_area_diario → tipo_equipo
          → equipo → empleado_equipo → proveedor → usuario_relacion
          → triggers e índices
============================================================ */

CREATE TABLE core.rol (
    ID_Rol  int IDENTITY(1,1) NOT NULL PRIMARY KEY,
    Nombre  varchar(50) NOT NULL,
    Activo  bit NOT NULL CONSTRAINT DF_core_rol_Activo DEFAULT (1),

    CONSTRAINT UQ_core_rol_Nombre UNIQUE (Nombre)
);
GO

CREATE TABLE core.permiso (
    ID_Permiso  int IDENTITY(1,1) NOT NULL PRIMARY KEY,
    Clave       varchar(50)  NOT NULL,
    Nombre      varchar(100) NOT NULL,
    Modulo      varchar(30)  NOT NULL,
    Descripcion varchar(200) NULL,
    Activo      bit NOT NULL CONSTRAINT DF_core_permiso_Activo DEFAULT (1),

    CONSTRAINT UQ_core_permiso_Clave UNIQUE (Clave)
);
GO

CREATE TABLE core.area (
    ID_Area int IDENTITY(1,1) NOT NULL PRIMARY KEY,
    Nombre  varchar(50) NOT NULL,
    Serie   varchar(3)  NOT NULL,
    Activo  bit NOT NULL CONSTRAINT DF_core_area_Activo DEFAULT (1),

    CONSTRAINT UQ_core_area_Nombre UNIQUE (Nombre),
    CONSTRAINT UQ_core_area_Serie  UNIQUE (Serie)
);
GO

CREATE TABLE core.puesto (
    ID_Puesto       int IDENTITY(1,1) NOT NULL PRIMARY KEY,
    Clave           varchar(9)   NOT NULL,
    Descripcion     varchar(100) NOT NULL,
    Nivel           int NOT NULL,
    Categoria       varchar(1)   NOT NULL,
    Segmento        varchar(3)   NOT NULL,
    Responsabilidad varchar(1)   NOT NULL,
    Activo          bit NOT NULL CONSTRAINT DF_core_puesto_Activo DEFAULT (1),

    CONSTRAINT UQ_core_puesto_Clave UNIQUE (Clave),
    CONSTRAINT CHK_core_puesto_Nivel CHECK (Nivel >= 0)
);
GO

CREATE TABLE core.sucursal (
    ID_Sucursal int NOT NULL PRIMARY KEY,   -- PK desde sistema externo (ERP)
    Nombre      varchar(100) NOT NULL,
    Ciudad      varchar(100) NULL,
    Activo      bit NOT NULL CONSTRAINT DF_core_suc_Activo DEFAULT (1),

    CONSTRAINT UQ_core_suc_Nombre UNIQUE (Nombre)
);
GO

CREATE TABLE core.empleado (
    Numero_Empleado   int NOT NULL PRIMARY KEY,   -- PK desde sistema externo
    Nombre            varchar(100) NOT NULL,
    Correo            varchar(100) NULL,
    Extension         varchar(10)  NULL,
    Telefono          varchar(20)  NULL,
    UsuarioAnyDesk    varchar(255) NULL,
    ClaveAnyDesk      varchar(255) NULL,
    ID_Sucursal       int NULL,
    ID_Puesto         int NULL,
    ID_Area           int NULL,
    Activo            bit NOT NULL CONSTRAINT DF_core_emp_Activo DEFAULT (1),
    FechaCreacion     datetime2(0) NOT NULL CONSTRAINT DF_core_emp_FC DEFAULT (SYSDATETIME()),
    FechaModificacion datetime2(0) NULL,

    CONSTRAINT FK_core_emp_sucursal FOREIGN KEY (ID_Sucursal) REFERENCES core.sucursal(ID_Sucursal),
    CONSTRAINT FK_core_emp_puesto   FOREIGN KEY (ID_Puesto)   REFERENCES core.puesto(ID_Puesto),
    CONSTRAINT FK_core_emp_area     FOREIGN KEY (ID_Area)     REFERENCES core.area(ID_Area)
);
GO

CREATE UNIQUE INDEX UQ_core_emp_Correo
    ON core.empleado(Correo)
    WHERE Correo IS NOT NULL;
GO

CREATE TABLE core.usuario (
    ID_Usuario        int IDENTITY(1,1) NOT NULL PRIMARY KEY,
    Nombre            varchar(50)  NOT NULL,
    Email             varchar(100) NOT NULL,
    FotoURL           varchar(255) NULL,
    ID_Rol            int NULL,
    ID_Area           int NULL,
    Activo            bit NOT NULL CONSTRAINT DF_core_usu_Activo DEFAULT (1),
    FechaCreacion     datetime2(0) NOT NULL CONSTRAINT DF_core_usu_FC DEFAULT (SYSDATETIME()),
    FechaModificacion datetime2(0) NULL,

    CONSTRAINT UQ_core_usu_Email UNIQUE (Email),
    CONSTRAINT FK_core_usu_rol  FOREIGN KEY (ID_Rol)  REFERENCES core.rol(ID_Rol),
    CONSTRAINT FK_core_usu_area FOREIGN KEY (ID_Area) REFERENCES core.area(ID_Area)
);
GO

/* --- TRIGGERS core --- */
CREATE OR ALTER TRIGGER core.TR_empleado_FM
ON core.empleado AFTER UPDATE
AS BEGIN
    SET NOCOUNT ON; IF TRIGGER_NESTLEVEL() > 1 RETURN;
    UPDATE e SET FechaModificacion = SYSDATETIME()
    FROM core.empleado e INNER JOIN inserted i ON e.Numero_Empleado = i.Numero_Empleado;
END;
GO

CREATE OR ALTER TRIGGER core.TR_usuario_FM
ON core.usuario AFTER UPDATE
AS BEGIN
    SET NOCOUNT ON; IF TRIGGER_NESTLEVEL() > 1 RETURN;
    UPDATE u SET FechaModificacion = SYSDATETIME()
    FROM core.usuario u INNER JOIN inserted i ON u.ID_Usuario = i.ID_Usuario;
END;
GO

END;
GO

CREATE TABLE hd.ticket (

    ID_Area_Origen      int NOT NULL,

    FechaReporte        datetime2(0) NOT NULL CONSTRAINT DF_hd_tkt_FR DEFAULT (SYSDATETIME()),
    FechaAsignacion     datetime2(0) NULL,
    FechaSolucion       datetime2(0) NULL,
    ID_Solucion         int NULL,

    ID_Soporte          int NULL,
    ID_Estatus          int NOT NULL,
    ID_Usuario_Cierra   int NULL,

    CONSTRAINT FK_hd_tkt_area_origen FOREIGN KEY (ID_Area_Origen)     REFERENCES core.area(ID_Area),
    CONSTRAINT FK_hd_tkt_area_resp  FOREIGN KEY (ID_Area_Responsable) REFERENCES core.area(ID_Area),
    CONSTRAINT FK_hd_tkt_sistema    FOREIGN KEY (ID_Sistema)          REFERENCES cat.sistema(ID_Sistema),
    CONSTRAINT FK_hd_tkt_error      FOREIGN KEY (ID_Error)            REFERENCES hd.[error](ID_Error),
    CONSTRAINT FK_hd_tkt_solucion   FOREIGN KEY (ID_Solucion)         REFERENCES hd.solucion(ID_Solucion),
    CONSTRAINT FK_hd_tkt_soporte    FOREIGN KEY (ID_Soporte)          REFERENCES core.usuario(ID_Usuario),
    CONSTRAINT FK_hd_tkt_estatus    FOREIGN KEY (ID_Estatus)          REFERENCES hd.estatus(ID_Estatus),
    CONSTRAINT FK_hd_tkt_cierra     FOREIGN KEY (ID_Usuario_Cierra)   REFERENCES core.usuario(ID_Usuario),
    CONSTRAINT CHK_hd_tkt_Fechas    CHECK (FechaAsignacion IS NULL OR FechaAsignacion >= FechaReporte),
    CONSTRAINT CHK_hd_tkt_FechaSol  CHECK (FechaSolucion IS NULL OR FechaAsignacion IS NULL
                                           OR FechaSolucion >= FechaAsignacion)
);
GO

/* --- TRIGGERS com --- */
CREATE OR ALTER TRIGGER com.TR_corrida_comision_FM
ON com.corrida_comision AFTER UPDATE
AS BEGIN
    SET NOCOUNT ON; IF TRIGGER_NESTLEVEL() > 1 RETURN;
    UPDATE c SET FechaModificacion = SYSDATETIME()
    FROM com.corrida_comision c INNER JOIN inserted i ON c.ID_Corrida = i.ID_Corrida;
END;
GO

CREATE OR ALTER TRIGGER com.TR_comision_base_FM
ON com.comision_base AFTER UPDATE
AS BEGIN
    SET NOCOUNT ON; IF TRIGGER_NESTLEVEL() > 1 RETURN;
    UPDATE c SET FechaModificacion = SYSDATETIME()
    FROM com.comision_base c INNER JOIN inserted i ON c.ID_Comision = i.ID_Comision;
END;
GO

CREATE OR ALTER TRIGGER com.TR_regla_comision_FM
ON com.regla_comision AFTER INSERT, UPDATE, DELETE
AS BEGIN
    SET NOCOUNT ON; IF TRIGGER_NESTLEVEL() > 1 RETURN;
    IF EXISTS (SELECT 1 FROM inserted) AND EXISTS (SELECT 1 FROM deleted)
    BEGIN
        UPDATE r SET FechaModificacion = SYSDATETIME()
        FROM com.regla_comision r INNER JOIN inserted i ON r.ID_Regla = i.ID_Regla;
    END;
    INSERT INTO com.regla_comision_historial
        (ID_Regla, PorcentajeMinimo, PorcentajeMaximo, Monto, Factor, Activo, Accion)
    SELECT d.ID_Regla, d.PorcentajeMinimo, d.PorcentajeMaximo, d.Monto, d.Factor, d.Activo,
           CASE WHEN EXISTS (SELECT 1 FROM inserted) THEN 'UPDATE' ELSE 'DELETE' END
    FROM deleted d;
    INSERT INTO com.regla_comision_historial
        (ID_Regla, PorcentajeMinimo, PorcentajeMaximo, Monto, Factor, Activo, Accion)
    SELECT i.ID_Regla, i.PorcentajeMinimo, i.PorcentajeMaximo, i.Monto, i.Factor, i.Activo, 'INSERT'
    FROM inserted i WHERE NOT EXISTS (SELECT 1 FROM deleted);
END;
GO
